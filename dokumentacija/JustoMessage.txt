CREATE SEQUENCE korisnik.PMSMESSAGE_SEQ
 START WITH     1
 INCREMENT BY   1
 NOCACHE
 NOCYCLE
/

CREATE TABLE korisnik.PMSMESSAGE (
      ID NUMBER(8) NOT NULL,
      POD NUMBER(3) NOT NULL,
      ORG NUMBER(3) NOT NULL,
      SOBA VARCHAR2(5) NOT NULL,
      PORUKA VARCHAR2(500) NOT NULL,
      STATUS NUMBER(1) DEFAULT 0 NOT NULL,
      VRIJEME_RCV DATE DEFAULT SYSDATE NOT NULL,
      OPER_ID_NOT NUMBER(5),
      VRIJEME_NOT DATE ,
      CONSTRAINT PK_PMSMESSAGE primary key (ID)) TABLESPACE korisnik
/
